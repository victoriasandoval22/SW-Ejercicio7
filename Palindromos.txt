    <script>
        function iniciar(){
            inverso = "";
            valor = document.getElementById("palabra").value;
            palabra = valor.length;
            for(var i=palabra; i>=0; i--){
                inverso = inverso + valor.charAt(i);
            }
    
            if(valor === inverso){
                alert(valor + ": es palindromo");
            }
            else{
                alert(valor + ": no es palindromo");
            }
        }
